249 inline document::doc_ref_result document::parser::parse(const uint8_t *buf, size_t len, bool realloc_if_needed) noexcept {
250  error_code code = init_parse(len);
251  if (code) { return document::doc_ref_result(doc, code); }
252 
253  if (realloc_if_needed) {
254  const uint8_t *tmp_buf = buf;
255  buf = (uint8_t *)allocate_padded_buffer(len);
256  if (buf == nullptr)
257  return document::doc_ref_result(doc, MEMALLOC);
258  memcpy((void *)buf, tmp_buf, len);
259  }
260 
261  code = simdjson::active_implementation->parse(buf, len, *this);
262 
263  // We're indicating validity via the doc_ref_result, so set the parse state back to invalid
264  valid = false;
265  error = UNINITIALIZED;
266  if (realloc_if_needed) {
267  aligned_free((void *)buf); // must free before we exit
268  }
269  return document::doc_ref_result(doc, code);
270 }
