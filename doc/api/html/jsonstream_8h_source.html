15 /*************************************************************************************
16  * The main motivation for this piece of software is to achieve maximum speed
17  *and offer
18  * good quality of life while parsing files containing multiple JSON documents.
19  *
20  * Since we want to offer flexibility and not restrict ourselves to a specific
21  *file
22  * format, we support any file that contains any valid JSON documents separated
23  *by one
24  * or more character that is considered a whitespace by the JSON spec.
25  * Namely: space, nothing, linefeed, carriage return, horizontal tab.
26  * Anything that is not whitespace will be parsed as a JSON document and could
27  *lead
28  * to failure.
29  *
30  * To offer maximum parsing speed, our implementation processes the data inside
31  *the
32  * buffer by batches and their size is defined by the parameter "batch_size".
33  * By loading data in batches, we can optimize the time spent allocating data in
34  *the
35  * parser and can also open the possibility of multi-threading.
36  * The batch_size must be at least as large as the biggest document in the file,
37  *but
38  * not too large in order to submerge the chached memory. We found that 1MB is
39  * somewhat a sweet spot for now. Eventually, this batch_size could be fully
40  * automated and be optimal at all times.
41  ************************************************************************************/

177 inline size_t find_last_json_buf_idx(const uint8_t *buf, size_t size,
178  const document::parser &parser) {
179  // this function can be generally useful
180  if (parser.n_structural_indexes == 0)
181  return 0;
182  auto last_i = parser.n_structural_indexes - 1;
183  if (parser.structural_indexes[last_i] == size) {
184  if (last_i == 0)
185  return 0;
186  last_i = parser.n_structural_indexes - 2;
187  }
188  auto arr_cnt = 0;
189  auto obj_cnt = 0;
190  for (auto i = last_i; i > 0; i--) {
191  auto idxb = parser.structural_indexes[i];
192  switch (buf[idxb]) {
193  case ':':
194  case ',':
195  continue;
196  case '}':
197  obj_cnt--;
198  continue;
199  case ']':
200  arr_cnt--;
201  continue;
202  case '{':
203  obj_cnt++;
204  break;
205  case '[':
206  arr_cnt++;
207  break;
208  }
209  auto idxa = parser.structural_indexes[i - 1];
210  switch (buf[idxa]) {
211  case '{':
212  case '[':
213  case ':':
214  case ',':
215  continue;
216  }
217  if (!arr_cnt && !obj_cnt) {
218  return last_i + 1;
219  }
220  return i;
221  }
222  return 0;
223 }

247  if (!allocok) {
248  return parser.error = simdjson::MEMALLOC;
249  }
250  } else if (unlikely(parser.capacity() < _batch_size)) {
251  return parser.error = simdjson::CAPACITY;
252  }
253  if (unlikely(parser_thread.capacity() < _batch_size)) {
254  const bool allocok_thread = parser_thread.allocate_capacity(_batch_size);
255  if (!allocok_thread) {
256  return parser.error = simdjson::MEMALLOC;
257  }
258  }
259  if (unlikely(load_next_batch)) {
260  // First time loading
261  if (!stage_1_thread.joinable()) {
262  _batch_size = (std::min)(_batch_size, remaining());
263  _batch_size = trimmed_length_safe_utf8((const char *)buf(), _batch_size);
264  if (_batch_size == 0) {
265  return parser.error = simdjson::UTF8_ERROR;
266  }
267  auto stage1_is_ok = error_code(simdjson::active_implementation->stage1(buf(), _batch_size, parser, true));
268  if (stage1_is_ok != simdjson::SUCCESS) {
269  return parser.error = stage1_is_ok;
270  }
271  size_t last_index = find_last_json_buf_idx(buf(), _batch_size, parser);
272  if (last_index == 0) {
273  if (parser.n_structural_indexes == 0) {
274  return parser.error = simdjson::EMPTY;
275  }
276  } else {
277  parser.n_structural_indexes = last_index + 1;
278  }
279  }
280  // the second thread is running or done.
281  else {
282  stage_1_thread.join();
283  if (stage1_is_ok_thread != simdjson::SUCCESS) {
284  return parser.error = stage1_is_ok_thread;
285  }
286  std::swap(parser.structural_indexes, parser_thread.structural_indexes);
287  parser.n_structural_indexes = parser_thread.n_structural_indexes;
288  advance(last_json_buffer_loc);
289  n_bytes_parsed += last_json_buffer_loc;
290  }
291  // let us decide whether we will start a new thread
292  if (remaining() - _batch_size > 0) {
293  last_json_buffer_loc =
294  parser.structural_indexes[find_last_json_buf_idx(buf(), _batch_size, parser)];
295  _batch_size = (std::min)(_batch_size, remaining() - last_json_buffer_loc);
296  if (_batch_size > 0) {
297  _batch_size = trimmed_length_safe_utf8(
298  (const char *)(buf() + last_json_buffer_loc), _batch_size);
299  if (_batch_size == 0) {
300  return parser.error = simdjson::UTF8_ERROR;
301  }
302  // let us capture read-only variables
303  const uint8_t *const b = buf() + last_json_buffer_loc;
304  const size_t bs = _batch_size;
305  // we call the thread on a lambda that will update
306  // this->stage1_is_ok_thread
307  // there is only one thread that may write to this value
308  stage_1_thread = std::thread([this, b, bs] {

310  });
311  }
312  }
313  next_json = 0;
314  load_next_batch = false;
315  } // load_next_batch
